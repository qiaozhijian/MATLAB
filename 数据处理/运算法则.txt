取模和取余
>> mod(5,2)
ans =1                   %“除数”是正，“余数”就是正
>> mod(-5,2)
ans =1
>> mod(5,-2)
ans =-1                  %“除数”是负，“余数‘就是负
>> mod(-5,-2)
ans =-1                  %用rem时，不管“除数”是正是负，“余数”的符号与“被除数”的符号相同
>> rem(5,2)
ans =1                   %“被除数”是正，“余数”就是正
>> rem(5,-2); 
ans =1
>> rem(-5,2)
ans =-1                 %“被除数”是负，“余数”就是负
>> rem(-5,-2)
ans =-1
